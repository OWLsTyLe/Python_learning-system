from django.views import View
from .models import Course
 
class CourseListView(View):
  def get(self, request):
    courses = Course.objects
      .filter(is_published=True)
      .order_by('-created_at')
    # Повертаємо JSON
    return JsonResponse({
      'courses': list(courses),
      'total': courses.count()
    })
